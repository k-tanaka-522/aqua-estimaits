# 2.5.3.2.4 TypeScript単体テストGood_Bad_Example集

## 目的

TypeScriptテストのよくある間違いと推奨パターンを示します。

---

## ❌ Bad Example

```typescript
// ❌ AAAパターンなし
test('get user', async () => {
  const result = await service.getUser(1);
  expect(result.name).toBe('Test');
});

// ❌ モックなし（実DB接続）
test('get user', async () => {
  const service = new UserService(new UserRepository()); // ❌ 実DB
  const result = await service.getUser(1);
});

// ❌ 型安全性なし（any使用）
test('get user', async () => {
  const mockRepo: any = { findById: jest.fn() }; // ❌ any
  const service = new UserService(mockRepo);
});

// ❌ エラーケースのテストなし
test('get user success only', async () => {
  // 成功ケースのみ、エラーケースなし ❌
  const result = await service.getUser(1);
  expect(result).toBeDefined();
});
```

---

## ✅ Good Example

```typescript
// ✅ AAAパターン + モック + 型安全性
describe('UserService', () => {
  let mockRepo: jest.Mocked<UserRepository>;
  let service: UserService;

  beforeEach(() => {
    mockRepo = {
      findById: jest.fn(),
      save: jest.fn(),
    } as jest.Mocked<UserRepository>;
    service = new UserService(mockRepo);
  });

  describe('getUser', () => {
    it('should return user when found', async () => {
      // Arrange
      const expectedUser: User = { id: 1, name: 'Test User', email: 'test@example.com' };
      mockRepo.findById.mockResolvedValue(expectedUser);

      // Act
      const result = await service.getUser(1);

      // Assert
      expect(result).toEqual(expectedUser);
      expect(mockRepo.findById).toHaveBeenCalledWith(1);
      expect(mockRepo.findById).toHaveBeenCalledTimes(1);
    });

    it('should throw error when user not found', async () => {
      // Arrange
      mockRepo.findById.mockResolvedValue(null);

      // Act & Assert
      await expect(service.getUser(999)).rejects.toThrow('User not found');
      expect(mockRepo.findById).toHaveBeenCalledWith(999);
    });

    it('should throw error when repository fails', async () => {
      // Arrange
      const dbError = new Error('Database connection failed');
      mockRepo.findById.mockRejectedValue(dbError);

      // Act & Assert
      await expect(service.getUser(1)).rejects.toThrow('Database connection failed');
    });
  });
});
```

### ✅ 型安全なモックの作成

```typescript
// ✅ 型安全なモック（Partial使用）
const mockRepo: jest.Mocked<Partial<UserRepository>> = {
  findById: jest.fn(),
};

// ✅ 完全な型定義
interface MockUserRepository {
  findById: jest.Mock<Promise<User | null>, [number]>;
  save: jest.Mock<Promise<User>, [User]>;
}

const mockRepo: MockUserRepository = {
  findById: jest.fn(),
  save: jest.fn(),
};
```
